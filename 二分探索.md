~~~cpp
int findBinarySearch(int val, vector<int>& v) {
	int l=0, r=v.size()-1;
	while(l <= r) {
		int m = l+(r-l)/2;
		if(val == v[m]) return m;
		else if(val < v[m]) l = m+1;
		else r = m-1;
	}
	return -1;
}
~~~
